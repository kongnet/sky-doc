// router/user.js /* global $ db redis */ module.exports = { async info(ctx) { const { userId } = ctx.checkedData.data const user = await db.user.find(userId) ctx.ok(user) } }

  • ctx.throw(status, msg) - 错误响应
  • ctx.$token - 当前用户

03

中间件示例

// middleware/auth.js module.exports = async (ctx, next) => { const token = ctx.headers.authorization if (!token) { ctx.throw(401, '请先登录') } const userInfo = await redis.get(`token:${token}`) ctx.$token = JSON.parse(userInfo) next && (await next()) }

05

MySQL - 事务

async transfer(fromId, toId, amount) { const conn = await db.transaction() try { await conn.user.decr(fromId, 'balance', amount) await conn.user.incr(toId, 'balance', amount) await conn.commit() return true } catch (err) { await conn.rollback() throw err } }

06

Redis - 实战

缓存用户信息

async getUser(id) { const cached = await redis.get(`u:${id}`) if (cached) return JSON.parse(cached) const user = await db.user.find(id) await redis.setex(`u:${id}`, 3600, JSON.stringify(user)) return user }

接口限流

async rateLimit(uid) { const key = `rl:${uid}` const n = await redis.incr(key) if (n === 1) redis.expire(key, 60) if (n > 100) ctx.throw(429, '太频繁') }
