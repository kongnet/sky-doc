$.groupBy / $.partition 分组

// 按函数分组 $.groupBy([1,2,3,4], x => x % 2) // → {0:[2,4], 1:[1,3]} // 分区 (满足/不满足) $.partition([1,2,3,4], x => x > 2) // → [[3,4], [1,2]]

$.debounce / $.throttle 防抖节流

$.debounce 防抖

// 等待 delay 后执行 const fn = $.debounce(() => { console.log('执行') }, 300) // 多次调用只在最后执行 fn(); fn(); fn(); // 只执行一次

$.throttle 节流

// 每隔 delay 执行一次 const fn = $.throttle(() => { console.log('执行') }, 1000) // 第一次立即执行 fn() // → 立即执行 fn() // → 忽略

$.retry / $.timeout 重试超时

$.once / $.memoize 缓存函数

// 只执行一次 const fn = $.once(() => { console.log('只执行一次') return Math.random() }) fn() // → 执行 fn() // → 返回第一次结果 fn() // → 返回第一次结果
// 记忆化缓存 const fib = $.memoize(n => { if (n <= 1) return n return fib(n-1) + fib(n-2) }) fib(40) // 快速返回 fib(40) // 使用缓存
